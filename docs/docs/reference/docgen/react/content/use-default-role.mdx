# `useDefaultRole()`

Gets the user's default role

```jsx
import { useUserRoles, useDefaultRole } from '@nhost/react'

const Avatar = () => {
  const roles = useUserRoles()
  const defaultRole = useDefaultRole()

  return (
    <div>
      Your default role is {defaultRole}. You have the following roles:{' '}
      {roles.join(', ')}
    </div>
  )
}
```
